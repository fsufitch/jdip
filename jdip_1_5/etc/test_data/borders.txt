#	These are some (but not comprehensive) tests of the

#############################################################
#	
# Set the Variant for all cases.
#
#############################################################
VARIANT_ALL TEST_Borders

CASE Borders.1 Season border test #1
PRESTATE_SETPHASE Spring 1901, Movement
PRESTATE
	England: A gas
ORDERS
	England: A gas-spa
POSTSTATE
	England: A gas
END

CASE Borders.2 Season border test #2
PRESTATE_SETPHASE Fall 1901, Movement
PRESTATE
	England: A gas
ORDERS
	England: A gas-spa
POSTSTATE
	England: A spa
END


CASE Borders.3 Phase border test #1
PRESTATE_SETPHASE Spring 1901, Movement
PRESTATE
	England: A mar
ORDERS
	England: A mar-pie
POSTSTATE
	England: A mar
END

CASE Borders.4 Phase border test #2
PRESTATE_SETPHASE Spring 1901, Retreat
PRESTATE
	England: A mar
PRESTATE_DISLODGED
	France: A mar
ORDERS
	England: A mar H
	France: A mar-pie
POSTSTATE
	England: A mar
	France:  A pie
END

CASE Borders.8 Order border test
PRESTATE_SETPHASE Spring 1901, Movement
PRESTATE
	England: A ber
	England: A ruhr
	France: A kiel
ORDERS
	England: A ber-kiel			# should fail
	England: A ruhr S ber H		# OK
	France: A kiel H
POSTSTATE
	England: A ber
	England: A ruhr
	France: A kiel
END
